$("form#uploadDataForm").submit(function(){

    var formData = new FormData(this);
	formData.append("lala","lolo");

    $.ajax({
        url: window.location.pathname,
        type: 'POST',
        data: formData,
        async: false,
        success: function (data) {
            location.reload(); // reload page
        },
        cache: false,
        contentType: false,
        processData: false
    });

    return false;
});
